import java.util.*;

class Main {
    public static void main(String[] args) {
        String str = "swiss";
        int n = str.length();
        char result = '-';

        for (int i = 0; i < n; i++) {
            if (str.indexOf(str.charAt(i)) == str.lastIndexOf(str.charAt(i))) {
                result = str.charAt(i);
                break;
            }
        }

        System.out.println("First Non-Repeating: " + result);
    }
}
