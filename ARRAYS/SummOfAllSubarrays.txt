class Main {
    public static void main(String[] args) {

        int[] arr = {1, 2, 3};
        int n = arr.length;
        int totalSum = 0;

        for (int i = 0; i < n; i++) {
            totalSum += arr[i] * (i + 1) * (n - i);
        }

        System.out.println("Sum of all subarrays: " + totalSum);
    }
}
